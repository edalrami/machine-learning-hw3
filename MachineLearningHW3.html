def get_centers(num, df):
    """
    Randomly pick n number of center points to be used for initial K-Means iteration
    
    Parameters:
    
    num: Desired number of cluster centers
    
    df: DataFrame containing positional data in x1 and x2 columns
    """
    #Randomly pick num center points within the dataset
    centers = [(random.uniform(df.x1.min(), df.x1.max()), random.uniform(df.x2.min(), df.x2.max()))for i in range(num)]
    return centers

def get_classes(data_point, center_points):
    """
    Classify a data point by computing its center to all n cluster centers.
    Point is classified to the nearest cluster.
    
    Parameters:
    
    data_point: A dataframe row with columns x1 and x2 that signify the data points position
    
    center_points: An array of tuples that contains the positional coordinates for all cluster centers
    """
    #compute distance to all 3 centers, return the min index as classification
    c = [ ((data_point.x1 - i[0])**2 + (data_point.x2 - i[1])**2)**0.5  for i in center_points]
    return np.argmin(c)

Look at that! We were able to classify each point with a certain class. We can now update the centers to be in the centroid of each class. The assignment requires us to update the centers only if the delta is less than 0.001. The process will be:

  1. Update the centers
  2. Calculate the delta
  3. Compare the previous two deltas
  4. Repeat if the conditions are not meet

def get_delta(prev_centers, new_centers):
    """
    Returns a list of n error values. Each error values signifies the square error of each
    cluster center from the previous K-Means iteration to the current iteration
    
    parameters: 
    
    prev_centers: array containing n tuples of n cluster centers from the previous K-Means iteration
    
    new_centers: array containing n tuples of n cluster centers from the current K-Means iteration
    """
    
    #compute error to previous centers
    dist = [((new_centers[i][0] - prev_centers[i][0])**2 + (new_centers[i][1] - prev_centers[i][1])**2) for i in range(len(new_centers))]
    
    #check that old and new centers are within 0.001 distance of each other
    err = [i < 0.001 for i in dist]
    return err

2¶

Train an RBF model using $\gamma = 0.5$. Report the correct classification rate of your model.

So! In order to solve for the RBF, we must define some variables. Let $\mu$ be the centers we found previously and $\gamma=0.5$. We can use the following estimation.

$$ \begin{bmatrix} 1 & e^{\gamma |X_{1} - \mu_{1} |^{2} } & e^{\gamma |X_{1} - \mu_{2} |^{2} } \\ 1 & e^{\gamma |X_{2} - \mu_{1} |^{2} } & e^{\gamma |X_{2} - \mu_{2} |^{2} } \\ \vdots & \vdots & \vdots \\ 1 & e^{\gamma |X_{N} - \mu_{1} |^{2} } & e^{\gamma |X_{N} - \mu_{2} |^{2} } \\ \end{bmatrix} \begin{bmatrix} w_{0} \\ w_{1} \\ w_{2} \end{bmatrix} \approx \begin{bmatrix} y_{0} \\ y_{1} \\ y_{2} \end{bmatrix} $$

To save space, we can rewrite the equation as:

$$ \Phi w \approx y \text{.} $$

Now we can solve for $w$ using linear regression:

$$ w \approx (\Phi^{T}\Phi)^{-1}\Phi^{T} y \text{.} $$

Alright! After solving for $w$, we can classify the points. We have 2 centers, $\mu$, and 3 weights, $w$. We can use $h(x)$ to classify each point. We must remember that the linear regression used a column of $1$'s, and we will substitute $1$ in the summation.

$$ h(x) = \sum_{k=1}^{K} w_{k} e^{-\gamma |x - \mu_{k} |^{2}} $$

After computing $h(x)$, we can look at the graph and see that some $h$ values are greater than 0.5. The ones greater than 0.5 will be classified as class 1, while the ones less than 0.5 are classified as class 0.

def computeDistances(categorical):
    """
    Returns a dictionary containing all vdm distances between all categories in the
    prob dictionary
    
    Parameters:
    
    categorical: Category key within the prob dictionary. Used to access the conditional
        probabilities in prob dictionary needed for VDM computations
    """
    not_cancel = categorical["not_cancel"]
    cancel = categorical["cancel"]
    cat_vals = cancel.keys()
    
    #Get the labels of the distance to be measured
    labels = [str(val) + "-" + str(val2) for val in cat_vals for val2 in cat_vals]
    
    #Compute all distances
    dist = [np.abs(cancel[val] - cancel[val2])**2 + np.abs(not_cancel[val] - not_cancel[val2])**2 
            for val in cat_vals for val2 in cat_vals]
    #Combine into a dictionary
    dist_dict = dict(zip(labels, dist))
    
    return dist_dict

def point_distance(point1, point2, params, distances):
    """
    Returns float value signifying the distance from 2 points with
    continuous and categorical features
    
    Parameters: 
    
    point1: A dataframe row signifying the first observation
    
    point2: A dataframe row signifying the second observation
    
    params: An array containg the categories to be iterated through in distance
    
    distance: A dictionary containing the VDM distances for the categorical variables in each point
    """
    #Separate categorical and continuous variables
    p1_cat = point1.values[2:]
    p2_cat = point2.values[2:]
    p1_con = point1.values[0:2]
    p2_con = point2.values[0:2]
    
    #compute continuous distance
    cont_dist = np.abs(p2_con[1] - p1_con[1])**2 + np.abs(p2_con[0] - p1_con[0])**2
    
    #Retrieve all the delta distance values
    dist_key = [str(p1_cat[i]) + "-" + str(p2_cat[i]) for i in range(len(p1_cat))]
    deltas = [distances[params[i]][dist_key[i]] for i in range(len(params))]
    
    #compute point distance with VDM
    res = np.power(cont_dist + sum(deltas), 0.5)
    return res

def within_cluster_error(df, k, distance_matrix, cat_params, cat_distances):
    """
    Clusters data points using AgglomerativeClustering() from sklearn and
    computes the within-cluster square errors for k clusters. Computes cluster centers
    and errors with VDM. 
    
    Returns a list dist containing the sum of square error for each cluster
    
    Parameters: 
    
    df: Dataframe containing all observations with continuous and categorical features
    
    k: Desired number of clusters
    
    distance_matrix: matrix containing all point distances from each observation to each observation.
        Used as input for AgglomerativeClustering()
        
    cat_params: A list containing the category keys in dictionary containing category VDM distances
    
    cat_distances: A dictionary containing the VDM distances between all categories 
    """
    
    
    
    clustering = AC(affinity='precomputed', compute_full_tree = True, linkage = 'complete', n_clusters = k).fit(distance_matrix)
    classifications = clustering.labels_
    #assign class to all points
    df['cls'] = classifications
    u, c =  np.unique(classifications, return_counts = True)
    
    centers = []
    for cls in list(df['cls'].unique()):
        cluster_x  = df[df['cls'] == cls]
        cont = list(cluster_x[0:2].mean(axis = 1))
        temp = []
        for ii in range(2,12):
            temp.append(cluster_x.iloc[:,ii].value_counts().index[0])
        res = cont + temp
        res = np.array(res)
        centers.append(res)
    
    centers_df = pd.DataFrame(centers, columns=list(df.columns))
    centers_df['avgYearPack'] = centers_df.avgYearPack.astype(float)
    centers_df['numYearAward'] = centers_df.numYearAward.astype(float)
    
    dist = []
    for center_index, center_row in centers_df.iterrows():
        temp = 0
        cls = int(center_row.cls)
        class_df = df[df['cls'] == cls]
        for class_index, class_row in class_df.iterrows():
            temp += point_distance(
                    center_row, 
                    class_row, 
                    cat_params, 
                    cat_distances)**2
        dist.append(temp)
    
    return dist
